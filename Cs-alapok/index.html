A c#-ban való programonzásokhoz való alapokat vettük át, mint a kiírás vagy a for-ciklusok. Itt van egy példa:

                    
                      static List testveresek_nevei(List lista, int testverek)
                          {
                              List result = new List();
                              foreach(Tanulo tanulo in lista)
                              {
                                  if (tanulo.testverek > testverek)
                                  {
                                      result.Add(tanulo);
                                  }
                              }
                              return result;
                          }
